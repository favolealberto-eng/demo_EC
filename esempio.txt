    <script>
      const forceUpdate = (id) => {
          const plane = document.getElementById(id);
          if(plane && plane.getObject3D('mesh')) plane.getObject3D('mesh').material.map.needsUpdate = true;
      };

      // ==========================================
      // GESTIONE UI, ZOOM E MODALE
      // ==========================================
      let activeMarker = null;
      let isPinned = false;
      let zoomLevels = {0: 1.0, 1: 1.0, 2: 1.0, 3: 1.0, 4: 1.0};

      function applyZoom(id) {
          const content = document.getElementById(`content-${id}`);
          if(content && !isPinned) {
              const z = zoomLevels[id];
              content.setAttribute('scale', `${z} ${z} ${z}`);
          }
      }

      function zoomPanel(dir) {
          if (activeMarker !== null && !isPinned) {
              zoomLevels[activeMarker] += dir;
              if(zoomLevels[activeMarker] < 0.4) zoomLevels[activeMarker] = 0.4;
              applyZoom(activeMarker);
          }
      }

      for(let i=0; i<=4; i++) {
          const marker = document.getElementById(`marker-${i}`);
          if(marker) {
              marker.addEventListener('markerFound', () => {
                  if (!isPinned) {
                      activeMarker = i;
                      document.getElementById('ui-controls').style.display = 'flex';
                      applyZoom(i); 
                  }
              });
              
              marker.addEventListener('markerLost', () => {
                  if (!isPinned && activeMarker === i) {
                      activeMarker = null;
                      document.getElementById('ui-controls').style.display = 'none';
                  }
              });
          }
      }

      function freezePanel() {
          if (activeMarker === null) return;
          isPinned = true;
          
          document.getElementById('ui-controls').style.display = 'none';
          
          const modal = document.getElementById('hud-modal');
          modal.style.display = 'flex';
          
          let canvasList = [];
          if (activeMarker === 0) canvasList.push('canvas-macchina');
          else if (activeMarker === 1) canvasList.push('canvas-iaq');
          else if (activeMarker === 2) canvasList.push('canvas-chart');
          else if (activeMarker === 3) canvasList.push('canvas-alarm');
          else if (activeMarker === 4) {
              canvasList.push('canvas-hot');
              canvasList.push('canvas-cold');
          }

          canvasList.forEach(id => {
              let cvs = document.getElementById(id);
              modal.appendChild(cvs);
              cvs.classList.add(activeMarker === 4 ? 'canvas-pinned-small' : 'canvas-pinned');
          });
      }

      function closeHud() {
          isPinned = false;
          const modal = document.getElementById('hud-modal');
          modal.style.display = 'none';
          
          const allCanvases = ['canvas-macchina', 'canvas-iaq', 'canvas-chart', 'canvas-alarm', 'canvas-hot', 'canvas-cold'];
          allCanvases.forEach(id => {
              let cvs = document.getElementById(id);
              document.body.appendChild(cvs);
              cvs.classList.remove('canvas-pinned');
              cvs.classList.remove('canvas-pinned-small');
          });
          
          const marker = document.getElementById(`marker-${activeMarker}`);
          if (marker && marker.object3D.visible) {
              document.getElementById('ui-controls').style.display = 'flex';
          } else {
              activeMarker = null;
          }
      }

      // ==========================================
      // LOGICA DATI E GRAFICA
      // ==========================================
      const ctxMacchina = document.getElementById('canvas-macchina').getContext('2d');
      const ctxIaq = document.getElementById('canvas-iaq').getContext('2d');

      function disegnaPannelloMacchina(dati) {
          const getVal = (nome) => { const item = dati.find(d => d.name === nome); return item ? item.value.toFixed(1) : "N/D"; };
          ctxMacchina.fillStyle = '#0f172a'; ctxMacchina.fillRect(0, 0, 1600, 900);
          ctxMacchina.fillStyle = '#ffffff'; ctxMacchina.font = 'bold 70px sans-serif'; ctxMacchina.fillText('CENTRALE TERMICA - JSON', 80, 120);

          function boxValore(ctx, x, y, tit, val, unita, col) {
              ctx.fillStyle = '#1e293b'; ctx.beginPath(); ctx.roundRect(x, y, 400, 300, 20); ctx.fill();
              ctx.fillStyle = '#94a3b8'; ctx.font = '35px sans-serif'; ctx.fillText(tit, x+40, y+70);
              ctx.fillStyle = col; ctx.font = 'bold 110px monospace'; ctx.fillText(val, x+40, y+200);
              ctx.fillStyle = '#94a3b8'; ctx.font = '40px sans-serif'; ctx.fillText(unita, x+280, y+200);
          }
          boxValore(ctxMacchina, 80, 250, "T. Mandata", getVal("Temperatura_Mandata_Primario"), "°C", "#ef4444");
          boxValore(ctxMacchina, 530, 250, "T. Ritorno", getVal("Temperatura_Ritorno_Primario"), "°C", "#3b82f6");
          boxValore(ctxMacchina, 980, 250, "Setpoint", getVal("SP_mandata_secondario"), "°C", "#10b981");
          forceUpdate('plane-macchina');
      }

      function disegnaPannelloIAQ(datiJson) {
          const signals = datiJson.message.signals;
          const getSig = (nome) => { const s = signals.find(s => s.name === nome); return s ? s.value : 0; };
          const co2 = getSig("co2");
          
          ctxIaq.fillStyle = '#0f172a'; ctxIaq.fillRect(0, 0, 1600, 900);
          ctxIaq.fillStyle = '#ffffff'; ctxIaq.font = 'bold 70px sans-serif'; ctxIaq.fillText('QUALITÀ ARIA - JSON', 80, 120);

          ctxIaq.fillStyle = '#cbd5e1'; ctxIaq.font = '40px sans-serif'; ctxIaq.fillText('Anidride Carbonica (CO2)', 80, 250);
          const grad = ctxIaq.createLinearGradient(80, 0, 1080, 0);
          grad.addColorStop(0, '#10b981'); grad.addColorStop(0.5, '#facc15'); grad.addColorStop(1, '#ef4444');
          ctxIaq.fillStyle = grad; ctxIaq.beginPath(); ctxIaq.roundRect(80, 280, 1000, 60, 30); ctxIaq.fill();
          
          let perc = (co2 - 400) / (2000 - 400); if(perc>1) perc=1; if(perc<0) perc=0;
          ctxIaq.fillStyle = '#ffffff'; ctxIaq.beginPath(); ctxIaq.arc(80 + (1000 * perc), 310, 20, 0, Math.PI*2); ctxIaq.fill();
          
          ctxIaq.font = 'bold 90px monospace'; ctxIaq.fillText(Math.round(co2), 1150, 340);
          ctxIaq.fillStyle = '#94a3b8'; ctxIaq.font = '40px sans-serif'; ctxIaq.fillText('ppm', 1400, 340);

          function boxM(x, y, tit, val, unita) {
              ctxIaq.fillStyle = '#1e293b'; ctxIaq.beginPath(); ctxIaq.roundRect(x, y, 420, 250, 20); ctxIaq.fill();
              ctxIaq.fillStyle = '#94a3b8'; ctxIaq.font = '35px sans-serif'; ctxIaq.fillText(tit, x+40, y+60);
              ctxIaq.fillStyle = '#ffffff'; ctxIaq.font = 'bold 90px monospace'; ctxIaq.fillText(val, x+40, y+170);
              ctxIaq.fillStyle = '#94a3b8'; ctxIaq.font = '40px sans-serif'; ctxIaq.fillText(unita, x+280, y+170);
          }
          boxM(80, 480, "Temperatura", getSig("temper").toFixed(1), "°C");
          boxM(550, 480, "Umidità", getSig("humid").toFixed(1), "%");
          boxM(1020, 480, "Polveri PM10", getSig("piemmedieci").toFixed(1), "µg");
          forceUpdate('plane-iaq');
      }

      async function scaricaJson() {
          const mockMacchina = [
              { "name": "SP_mandata_secondario", "value": 18.6 }, { "name": "Temperatura_Mandata_Primario", "value": 26.1 },
              { "name": "Temperatura_Ritorno_Primario", "value": 25.5 }, { "name": "Temperatura_Mandata_Caldaia_1", "value": 26.4 }
          ];
          const mockIaq = { "message": { "signals": [{ "name": "co2", "value": 687 }, { "name": "temper", "value": 25.12 }, { "name": "humid", "value": 33.46 }, { "name": "piemmedieci", "value": 1.4 }]}};
          try { const r = await fetch('./dati-endpoint.json'); if(!r.ok) throw new Error(); disegnaPannelloMacchina(await r.json()); } catch (e) { disegnaPannelloMacchina(mockMacchina); }
          try { const r = await fetch('./messagio-iaq.json'); if(!r.ok) throw new Error(); disegnaPannelloIAQ(await r.json()); } catch (e) { disegnaPannelloIAQ(mockIaq); }
      }
      setInterval(scaricaJson, 3000); scaricaJson();

      const ctxChart = document.getElementById('canvas-chart').getContext('2d');
      Chart.defaults.color = '#cbd5e1'; Chart.defaults.font.family = 'sans-serif';
      const chartInstance = new Chart(ctxChart, {
          type: 'line',
          data: {
              labels: ['-10s', '-8s', '-6s', '-4s', '-2s', 'Ora'],
              datasets: [{ label: 'Potenza kW', data: [15.2, 15.5, 16.1, 15.8, 16.5, 16.2], borderColor: '#0ea5e9', backgroundColor: 'rgba(14, 165, 233, 0.2)', borderWidth: 6, tension: 0.4, fill: true, pointRadius: 8, pointBackgroundColor: '#ffffff' }]
          },
          options: { responsive: false, animation: { duration: 500 }, layout: { padding: 40 }, plugins: { customCanvasBackgroundColor: true, legend: { labels: { font: { size: 24 } } }, title: { display: true, text: 'POTENZA MACCHINA', color: '#ffffff', font: { size: 64, weight: 'bold' } }, subtitle: { display: true, text: 'Live Data Simulato', color: '#94a3b8', font: { size: 32 }, padding: { bottom: 40 } } }, scales: { y: { min: 10, max: 25, ticks: { font: { size: 24 } } }, x: { ticks: { font: { size: 24 } } } } },
          plugins: [{ id: 'customCanvasBackgroundColor', beforeDraw: (ch) => { const c = ch.canvas.getContext('2d'); c.save(); c.globalCompositeOperation = 'destination-over'; c.fillStyle = '#0f172a'; c.fillRect(0, 0, ch.width, ch.height); c.restore(); } }]
      });

      setInterval(() => {
          chartInstance.data.datasets[0].data.shift(); chartInstance.data.datasets[0].data.push((Math.random() * 4 + 14).toFixed(1));
          chartInstance.options.plugins.subtitle.text = `Aggiornato: ${new Date().toLocaleTimeString()}`; chartInstance.update(); forceUpdate('plane-chart');
      }, 2000);

      const ctxAlarm = document.getElementById('canvas-alarm').getContext('2d');
      let tAlarm = 65.0, upAlarm = true;
      setInterval(() => { if (upAlarm) { tAlarm += Math.random() * 3; if (tAlarm > 95) upAlarm = false; } else { tAlarm -= Math.random() * 2; if (tAlarm < 65) upAlarm = true; } }, 1000);

      function loopAlarm() {
          ctxAlarm.fillStyle = '#0f172a'; ctxAlarm.fillRect(0, 0, 1600, 900);
          let st, cB, cD, tSt, tDet;
          if (tAlarm <= 75) { st=0; cB='#10b981'; cD='#10b981'; tSt="SISTEMA OK"; tDet=`Valore: ${tAlarm.toFixed(1)} °C`; }
          else if (tAlarm <= 85) { st=1; cB='#facc15'; cD='#713f12'; tSt="ANOMALIA"; tDet=`Attuale: ${tAlarm.toFixed(1)} °C | Max: 75.0 °C`; }
          else { st=2; cB='#ef4444'; cD='#7f1d1d'; tSt="CRITICO"; tDet=`Attuale: ${tAlarm.toFixed(1)} °C | OVERHEAT`; }

          const blink = (Date.now() % 1000) < 500; const colSpia = (st > 0 && !blink) ? cD : cB;
          ctxAlarm.beginPath(); ctxAlarm.arc(250, 450, 120, 0, 2 * Math.PI); ctxAlarm.shadowBlur = (st > 0 && !blink) ? 0 : 40; ctxAlarm.shadowColor = cB;
          ctxAlarm.fillStyle = colSpia; ctxAlarm.fill(); ctxAlarm.shadowBlur = 0; ctxAlarm.lineWidth = 15; ctxAlarm.strokeStyle = cB; ctxAlarm.stroke();
          ctxAlarm.fillStyle = '#ffffff'; ctxAlarm.font = 'bold 70px sans-serif'; ctxAlarm.fillText('DIAGNOSTICA', 100, 150);
          ctxAlarm.fillStyle = cB; ctxAlarm.font = 'bold 110px sans-serif'; ctxAlarm.fillText(tSt, 450, 460);
          ctxAlarm.fillStyle = '#cbd5e1'; ctxAlarm.font = '55px monospace'; ctxAlarm.fillText(tDet, 450, 560);
          forceUpdate('plane-alarm'); requestAnimationFrame(loopAlarm);
      }
      requestAnimationFrame(loopAlarm);

      const ctxHot = document.getElementById('canvas-hot').getContext('2d'), ctxCold = document.getElementById('canvas-cold').getContext('2d');
      const arrHot = document.querySelectorAll('.arrow-hot'), arrCold = document.querySelectorAll('.arrow-cold');
      let tHot = 95.0, tCold = 7.0;
      function colora(c1, c2, f) { return `rgb(${Math.round(c1[0]+f*(c2[0]-c1[0]))}, ${Math.round(c1[1]+f*(c2[1]-c1[1]))}, ${Math.round(c1[2]+f*(c2[2]-c1[2]))})`; }

      setInterval(() => {
          tHot += (Math.random() * 2 - 1); tCold += (Math.random() * 0.6 - 0.3);
          ctxHot.clearRect(0, 0, 1024, 256); ctxHot.fillStyle = '#0f172a'; ctxHot.beginPath(); ctxHot.roundRect(10, 10, 1004, 236, 40); ctxHot.fill();
          ctxHot.textAlign = 'center'; ctxHot.fillStyle = '#ffb74d'; ctxHot.font = 'bold 50px sans-serif'; ctxHot.fillText('MANDATA VAPORE', 512, 90);
          ctxHot.fillStyle = '#ffffff'; ctxHot.font = 'bold 110px monospace'; ctxHot.fillText(`${tHot.toFixed(1)} °C`, 512, 210);
          ctxCold.clearRect(0, 0, 1024, 256); ctxCold.fillStyle = '#0f172a'; ctxCold.beginPath(); ctxCold.roundRect(10, 10, 1004, 236, 40); ctxCold.fill();
          ctxCold.textAlign = 'center'; ctxCold.fillStyle = '#4fc3f7'; ctxCold.font = 'bold 50px sans-serif'; ctxCold.fillText('RITORNO ACQUA', 512, 90);
          ctxCold.fillStyle = '#ffffff'; ctxCold.font = 'bold 110px monospace'; ctxCold.fillText(`${tCold.toFixed(1)} °C`, 512, 210);
          forceUpdate('plane-hot'); forceUpdate('plane-cold');
          let fH = Math.max(0, Math.min(1, (tHot - 85) / 20)); let cH = colora([255, 165, 0], [255, 0, 0], fH); arrHot.forEach(a => a.setAttribute('color', cH));
          let fC = Math.max(0, Math.min(1, (tCold - 2) / 10)); let cC = colora([0, 0, 255], [0, 255, 255], fC); arrCold.forEach(a => a.setAttribute('color', cC));
      }, 500);
    </script>
